          setTimeout(typeWriter, 50);
        }
      };
      
      const quoteObserver = new IntersectionObserver((entries) => {
        if (entries[0].isIntersecting) {
          typeWriter();
          quoteObserver.unobserve(quote);
        }
      }, { threshold: 0.5 });
      
      quoteObserver.observe(quote);
    }
  });
</script>
